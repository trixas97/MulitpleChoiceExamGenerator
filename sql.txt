DROP TABLE users;
DROP TABLE lessons;
DROP TABLE units;
DROP TABLE questions;
DROP TABLE answers;
DROP TABLE tests;


CREATE TABLE users(
    username varchar(90) primary key,
    password varchar(90),
    surname varchar(90),
    name varchar(90),
    email varchar(90),
    school varchar(90),
    gender varchar(90),
    birth varchar(90));

CREATE TABLE lessons(

    name varchar(90),

    owner varchar(90),

    PRIMARY KEY(name,owner),

    CONSTRAINT fk_lessons_users

        FOREIGN KEY (owner) REFERENCES users(username)

        ON DELETE CASCADE

	ON UPDATE CASCADE);

CREATE TABLE units(

    id int primary key AUTO_INCREMENT,

    name varchar(90),

    owner varchar(90),

    lesson varchar(90),

    CONSTRAINT fk_un_lessons

        FOREIGN KEY (lesson,owner) REFERENCES lessons(name,owner)

        ON DELETE CASCADE

	ON UPDATE CASCADE);

CREATE TABLE questions(

    id int primary key AUTO_INCREMENT,

    text varchar(90),

    owner varchar(90),

    lesson varchar(90),

    unit_id int,

    dif int,

    CONSTRAINT fk_que_lessons

        FOREIGN KEY (lesson,owner) REFERENCES lessons(name,owner)

        ON DELETE CASCADE
	ON UPDATE CASCADE,

    CONSTRAINT fk_que_un

        FOREIGN KEY (unit_id) REFERENCES units(id)

        ON DELETE CASCADE

	ON UPDATE CASCADE);

CREATE TABLE answers(

    id int primary key AUTO_INCREMENT,

    text varchar(90),

    id_q int,

    correct boolean,

    CONSTRAINT fk_ans_que

        FOREIGN KEY (id_q) REFERENCES questions(id)

        ON DELETE CASCADE
	ON UPDATE CASCADE
);

CREATE TABLE tests(

    id int,

    id_q int,

    owner varchar(90),

    unit_id int,

    lesson varchar(90),

    primary key(id,id_q),

    CONSTRAINT fk_test_que

	FOREIGN KEY (id_q) REFERENCES questions(id)

        ON DELETE CASCADE
	ON UPDATE CASCADE,

    CONSTRAINT fk_test_user

	FOREIGN KEY (owner) REFERENCES users(username)

        ON DELETE CASCADE
	ON UPDATE CASCADE,

    CONSTRAINT fk_test_lesson

	FOREIGN KEY (lesson) REFERENCES lessons(name)

        ON DELETE CASCADE
	ON UPDATE CASCADE);
